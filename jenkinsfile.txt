pipeline {
  agent{
       docker{
         image 'nodejs:12.4.0-npm'
       }

    }
    
  stages {
        
    stage('Cloning Git') {
      steps {
        git 'https://github.com/cmad-team5/device-ui.git'
      }
    }
        
    stage('Install dependencies') {
      steps {
        sh 'npm install'
      }
    }
     
    stage('Test') {
      steps {
         sh 'npm test'
      }
    }      
  }
}
